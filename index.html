// Função para gerar imagens PNG dos QR codes lidos function generateQrCodeImages() { qrCodes.forEach(function(qrCode, index) { // Cria um elemento temporário para renderizar o QR code let canvas = document.createElement('canvas'); let context = canvas.getContext('2d'); // Configura o tamanho do canvas canvas.width = 200; // Largura canvas.height = 200; // Altura // Cria um objeto QRious com o conteúdo do QR code atual let qr = new QRious({ element: canvas, value: qrCode, size: 200 // Tamanho do QR code }); // Cria um link de download para o PNG do QR code let link = document.createElement('a'); link.href = canvas.toDataURL('image/png'); // Converte o canvas para data URL PNG link.download = `qr_code_${index + 1}.png`; // Nome do arquivo para download link.textContent = `QR Code ${index + 1}`; // Texto do link // Cria um parágrafo

para exibir o conteúdo do QR code let paragraph = document.createElement('p'); paragraph.textContent = `Conteúdo do QR Code ${index + 1}: ${qrCode}`; // Adiciona a imagem e o parágrafo à página document.body.appendChild(link); document.body.appendChild(paragraph); document.body.appendChild(document.createElement('br')); // Quebra de linha para separar os elementos }); }
